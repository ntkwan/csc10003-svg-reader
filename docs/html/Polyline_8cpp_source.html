57  int n = edges.size();
58

60  for (int i = 0; i < points.size(); i++) {
61  _points.push_back({points[i], 0});
62  }
63  for (int i = 0; i < n - 1; i++) {
64  float x1 = edges[i].start.x;
65  float y1 = edges[i].start.y;
66  float x2 = edges[i].end.x;
67  float y2 = edges[i].end.y;
68 
69  float x3 = edges[n - 1].start.x;
70  float y3 = edges[n - 1].start.y;
71  float x4 = edges[n - 1].end.x;
72  float y4 = edges[n - 1].end.y;
73 
74  float det = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4);
75 
76  if (det == 0) continue;
77 
78  float t = ((x1 - x3) * (y3 - y4) - (y1 - y3) * (x3 - x4)) /
79  ((x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4));
80  float intersectionX = x1 + t * (x2 - x1);
81  float intersectionY = y1 + t * (y2 - y1);
82

84  PolygonPoint intersectionPolygonPoint{intersectionPoint, 1};

91  }

122  float y2 = edge1.end.y;
123

133  }
134 
135  float t = ((x1 - x3) * (y3 - y4) - (y1 - y3) * (x3 - x4)) /

137 
138  float intersectionX = x1 + t * (x2 - x1);
139  float intersectionY = y1 + t * (y2 - y1);

164 void Polyline::draw(sf::RenderWindow& target, sf::RenderStates states) const {
165  if (points.size() < 2) return;
166  sf::VertexArray lineStrip(sf::PrimitiveType::Quads);
167  sf::Vector2f p1a, p1b, p2a, p2b;
168  sf::Vector2f r_p1a, r_p1b, r_p2a, r_p2b;
169  for (std::size_t i = 1; i < points.size(); i++) {
170  sf::Vector2f p1 = points[i - 1];
171  sf::Vector2f p2 = points[i];
172 
173  sf::Vector2f delta = p2 - p1;
174  float length = std::sqrt(delta.x * delta.x + delta.y * delta.y);
175 
176  sf::Vector2f unitDirection = delta / length;
177 
178  sf::Vector2f perpendicularDirection(-unitDirection.y, unitDirection.x);
179 
180  float thickness = getOutlineThickness();
181  sf::Color stroke = getOutlineColor();
182 
183  p1a = p1 - perpendicularDirection * (thickness / 2.0f);
184  p1b = p1 + perpendicularDirection * (thickness / 2.0f);
185  p2a = p2 - perpendicularDirection * (thickness / 2.0f);
186  p2b = p2 + perpendicularDirection * (thickness / 2.0f);
187  if (i > 1) {
188  sf::VertexArray lS(sf::PrimitiveType::Quads);
189  if (isPerpendicular({points[i], points[i - 1]},
190  {points[i - 1], points[i - 2]})) {
191  lS.append(sf::Vertex(
192  findIntersection({r_p1a, r_p2a}, {p1a, p2a}), stroke));
193  lS.append(sf::Vertex(
194  findIntersection({r_p1a, r_p2a}, {p1b, p2b}), stroke));
195  lS.append(sf::Vertex(
196  findIntersection({r_p1b, r_p2b}, {p1b, p2b}), stroke));
197  lS.append(sf::Vertex(
198  findIntersection({r_p1b, r_p2b}, {p1a, p2a}), stroke));
199 
200  } else {
201  lS.append(sf::Vertex(p1a, stroke));
202  lS.append(sf::Vertex(r_p2a, stroke));
203  lS.append(sf::Vertex(p1b, stroke));
204  lS.append(sf::Vertex(r_p2b, stroke));
205  }
206  target.draw(lS);
207  }
208  r_p1a = p1a;
209  r_p1b = p1b;
210  r_p2a = p2a;
211  r_p2b = p2b;
212  lineStrip.append(sf::Vertex(p1a, stroke));
213  lineStrip.append(sf::Vertex(p1b, stroke));
214  lineStrip.append(sf::Vertex(p2b, stroke));
215  lineStrip.append(sf::Vertex(p2a, stroke));
216  }
217  std::vector< ClosedPolygon > cP = findClosedPolygons(points);
218  if (cP.size() > 0) {
219  sf::Color fillColor = getFillColor();
220  for (const ClosedPolygon& polygon : cP) {
221  if (polygon.cP.size() > 2) { // Ensure it's a valid polygon
222  sf::ConvexShape fillShape;
223  fillShape.setFillColor(fillColor);
224  fillShape.setOutlineThickness(0);
225  fillShape.setPointCount(polygon.cP.size());
226 
227  // Set the points of the shape based on the polygon
228  for (std::size_t j = 0; j < polygon.cP.size(); j++) {
229  fillShape.setPoint(j, polygon.cP[j]);
230  }
231 
232  target.draw(fillShape);
233  }
234  }
235  target.draw(lineStrip);
236  }
